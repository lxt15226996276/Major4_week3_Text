### 3.1 LoginController · §11.9 三块

#### 本段思路

1. 读 `accountInput.text` / `passwordInput.text`，**Trim** 去空格。  
2. 与 **`const` 自定义账号密码** 比较（卷面允许自定；仓库建议 `lixiaotong` / `123456`）。  
3. 失败 → **`Debug.Log` 失败文案**；成功 → Log 成功 + **`LoadScene(Exam01SceneNames.Main)`**。  
4. **`Start` 代码绑 Button** + 可选 **`onSubmit`**；**`OnDestroy` RemoveListener**。

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

namespace Exam.Exam01
{
    /// <summary>W3 Exam01 · 登录校验并跳转主界面</summary>
    public class LoginController : MonoBehaviour
    {
        // --- 卷面自定义账号密码（考试可改常量） ---
        private const string ValidAccount = "lixiaotong";
        private const string ValidPassword = "123456";

        [Header("UI")]
        [SerializeField] private InputField _accountInput;
        [SerializeField] private InputField _passwordInput;
        [SerializeField] private Button _loginButton;
        [SerializeField] private Text _tipText; // 可选 · P-L1

        private void Start()
        {
            _loginButton.onClick.AddListener(OnLoginClick);
            // P-L1：回车链（可选）
            _accountInput.onSubmit.AddListener(_ => _passwordInput.ActivateInputField());
            _passwordInput.onSubmit.AddListener(_ => OnLoginClick());
        }

        /// <summary>试卷：失败 Console · 成功跳主界面</summary>
        public void OnLoginClick()
        {
            string account = _accountInput.text.Trim();
            string password = _passwordInput.text.Trim();

            if (account == ValidAccount && password == ValidPassword)
            {
                Debug.Log("登录成功");
                SceneManager.LoadScene(Exam01SceneNames.Main);
            }
            else
            {
                Debug.Log("登录失败");
                if (_tipText != null) _tipText.text = "账号或密码错误";
            }
        }

        private void OnDestroy()
        {
            if (_loginButton != null)
                _loginButton.onClick.RemoveListener(OnLoginClick);
        }
    }
}
```

### 6.2 SkillListController + PlayerSkillAnimator · §11.9

#### 本段思路

1. 点 **BtnSkill** → 显示 **SkillPanel**（MainUIController）。  
2. 点 **某个技能钮** → 显示 **SkillDetailPanel** + 填 Title/Desc。  
3. 同时调 **`PlayerSkillAnimator.PlaySkill(index)`** → **`SetTrigger`**。  
4. **SkillDetailPanel** 与 **SkillPanel** 各有 **关闭钮**。

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.UI;

namespace Exam.Exam01
{
    public class PlayerSkillAnimator : MonoBehaviour
    {
        [SerializeField] private Animator _animator;

        private static readonly int AttackHash = Animator.StringToHash("Attack");
        private static readonly int SkillUHash = Animator.StringToHash("SkillU");
        private static readonly int SkillIHash = Animator.StringToHash("SkillI");

        public void PlaySkill(int skillIndex)
        {
            switch (skillIndex)
            {
                case 0: _animator.SetTrigger(AttackHash); break;
                case 1: _animator.SetTrigger(SkillUHash); break;
                case 2: _animator.SetTrigger(SkillIHash); break;
                case 3: _animator.SetTrigger(AttackHash); break; // L1 复用 Attack
            }
        }
    }

    public class SkillListController : MonoBehaviour
    {
        [System.Serializable]
        public struct SkillInfo
        {
            public string displayName;
            [TextArea] public string description;
        }

        [SerializeField] private SkillInfo[] _skills = new SkillInfo[4];
        [SerializeField] private Button[] _skillButtons;
        [SerializeField] private GameObject _detailPanel;
        [SerializeField] private Text _detailTitle;
        [SerializeField] private Text _detailDesc;
        [SerializeField] private PlayerSkillAnimator _playerAnimator;

        private void Start()
        {
            for (int i = 0; i < _skillButtons.Length; i++)
            {
                int index = i; // 闭包捕获
                _skillButtons[i].onClick.AddListener(() => OnSkillSelected(index));
            }
        }

        private void OnSkillSelected(int index)
        {
            _detailPanel.SetActive(true);
            _detailTitle.text = _skills[index].displayName;
            _detailDesc.text = _skills[index].description;
            _playerAnimator.PlaySkill(index);
        }

        public void CloseDetailPanel() => _detailPanel.SetActive(false);
        public void CloseSkillPanel() => gameObject.SetActive(false);
    }
}
```

#### 逐行讲解表

| 行/块 | 含义 |
|-------|------|
| `StringToHash` | 对齐 Master.controller · 防拼写 |
| `int index = i` | foreach/for 闭包 **必写** |
| `_detailPanel.SetActive(true)` | **提示弹窗** 展示基本信息 |
| `PlaySkill` | 试卷 **主角展示对应动作** |
