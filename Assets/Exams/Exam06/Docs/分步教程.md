### 3.1 LoadingController

| 步骤 | 操作 |
|------|------|
| 1 | 新建 `Assets/Exams/Exam06/Scripts/LoadingController.cs` · `namespace Exam.Exam06` |
| 2 | 挂到 LoadingCanvas 或空物体 |

#### 本段思路

1. **`duration = 6f`**（Exam05 是 5f · **本套核心差异**）。  
2. 协程刷 `progressSlider.value` + 上下 Text。  
3. 静音：`MuteBtnOnClick` **命名方法** + `OnDestroy` **RemoveListener**。  
4. **P+1**：`LoadSceneAsync` + **`Min(fake, real/0.9f)`**（继承 Exam05 · 禁止 Max）。  
5. **P-L1（v3）**：**`Start` 启协程**（禁止仅 `OnEnable` 无停 · **L3**）· Play 从 **本套 Loading**（**L2**）。

#### 带注释 P-L1 范本

```csharp
using System.Collections;
using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;

namespace Exam.Exam06
{
    /// <summary>
    /// Exam06 加载：6 秒进度条 + 上下提示 + 静音，完成后进入 GameScene。
    /// </summary>
    public class LoadingController : MonoBehaviour
    {
        // --- UI 引用 ---
        [SerializeField] private Slider progressSlider;
        [SerializeField] private Text tipProgressText;
        [SerializeField] private Text tipTextBottom;
        [SerializeField] private Button muteBtn;
        [SerializeField] private AudioSource audioSource;

        // --- 加载参数 ---
        [SerializeField] private float duration = 6f; // Exam06：6 秒（Exam05 为 5）
        private readonly string nextScene = "Exam06_GameScene";

        /// <summary>
        /// 思路：绑静音 → 启动加载协程。
        /// </summary>
        private void Start()
        {
            muteBtn.onClick.AddListener(MuteBtnOnClick);
            StartCoroutine(LoadGameRoutine());
        }

        /// <summary>
        /// 思路：切换 mute → 可选更新钮文字/图标。
        /// </summary>
        private void MuteBtnOnClick()
        {
            audioSource.mute = !audioSource.mute;
        }

        /// <summary>
        /// 思路：Async 预加载 + 6s 时间轴 → Min 双轨写 Slider → 激活场景。
        /// </summary>
        private IEnumerator LoadGameRoutine()
        {
            var op = SceneManager.LoadSceneAsync(nextScene);
            op.allowSceneActivation = false;

            float elapsed = 0f;
            while (elapsed < duration)
            {
                elapsed += Time.deltaTime;
                float fake = elapsed / duration;
                float real = op.progress / 0.9f;
                float progress = Mathf.Clamp01(Mathf.Min(fake, real)); // 禁止 Max
                progressSlider.value = progress;
                tipProgressText.text = "加载中 " + (int)(progress * 100) + "%";
                tipTextBottom.text = GetTipByProgress(progress);
                yield return null;
            }

            progressSlider.value = 1f;
            tipProgressText.text = "加载中 100%";
            yield return new WaitForSeconds(0.2f);
            op.allowSceneActivation = true;
        }

        /// <summary>思路：按进度段换底部提示句。</summary>
        private string GetTipByProgress(float t)
        {
            if (t < 0.33f) return "正在连接服务器...";
            if (t < 0.66f) return "正在加载场景...";
            return "即将进入游戏...";
        }

        /// <summary>思路：对称解绑，防重复进场景委托残留。</summary>
        private void OnDestroy()
        {
            if (muteBtn != null)
                muteBtn.onClick.RemoveListener(MuteBtnOnClick);
        }
    }
}
```

#### 4.2.4 `Horizontal Layout Group` · 逐字段

| 字段 | 值 | 为何 |
|------|-----|------|
| Padding | **10** | 不贴边 |
| Spacing | **24** ★ | **禁止 250** — 你仓库现状会挤满屏 |

| Control Child Size | Width **☐** · Height **☐** | 不拉伸钮图 |
| Child Force Expand | Width **☐** · Height **☐** ★ | **必须关** — 开=挤出/拉变形 |
| Use Child Scale | **☐** | 不用 Scale 参与布局 |

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.UI;

namespace Exam.Exam06
{
    /// <summary>
    /// Exam06 战斗：三钮扣血 20/30/50，hp≤0 显示 GameOverPanel 并 Log。
    /// </summary>
    public class BattleController : MonoBehaviour
    {
        // --- UI 引用 ---
        [SerializeField] private Slider hpSlider;
        [SerializeField] private Text hpLabel;
        [SerializeField] private GameObject gameOverPanel;

        // --- 攻击钮 · 命名方法绑定 ---
        [SerializeField] private Button btnAttack20;
        [SerializeField] private Button btnAttack30;
        [SerializeField] private Button btnAttack50;

        // --- 战斗参数 ---
        [SerializeField] private int maxHp = 100;
        private int currentHp;
        /// <summary>结束后忽略扣血输入。</summary>
        private bool isGameOver;

        /// <summary>
        /// 思路：初始化 HP/Slider → 隐藏 Panel → 绑三钮命名回调。
        /// </summary>
        private void Start()
        {
            currentHp = maxHp;
            hpSlider.maxValue = maxHp;
            hpSlider.value = currentHp;
            gameOverPanel.SetActive(false);
            RefreshHpText();

            btnAttack20.onClick.AddListener(OnAttack20);
            btnAttack30.onClick.AddListener(OnAttack30);
            btnAttack50.onClick.AddListener(OnAttack50);
        }

        private void OnAttack20() => TakeDamage(20);
        private void OnAttack30() => TakeDamage(30);
        private void OnAttack50() => TakeDamage(50); // 第三钮 50 · 非 40

        /// <summary>
        /// 思路：门闩 → Clamp 扣血 → 刷 UI → hp≤0 结束态。
        /// </summary>
        private void TakeDamage(int damage)
        {
            if (isGameOver) return;
            currentHp = Mathf.Clamp(currentHp - damage, 0, maxHp);
            hpSlider.value = currentHp; // §11.20：只 value · 不 Hide Fill
            RefreshHpText();
            if (currentHp <= 0)
            {
                isGameOver = true;
                gameOverPanel.SetActive(true);
                Debug.Log("游戏结束");
            }
        }

        private void RefreshHpText()
        {
            if (hpLabel != null)
                hpLabel.text = "生命值：" + currentHp;
        }

        private void OnDestroy()
        {
            if (btnAttack20 != null) btnAttack20.onClick.RemoveListener(OnAttack20);
            if (btnAttack30 != null) btnAttack30.onClick.RemoveListener(OnAttack30);
            if (btnAttack50 != null) btnAttack50.onClick.RemoveListener(OnAttack50);
        }
    }
}
```
