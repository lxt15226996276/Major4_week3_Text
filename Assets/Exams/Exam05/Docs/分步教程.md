### 3.1 进度协程 + 上下文字联动

#### 本段思路

1. **`loadDuration = 5f`**：本套固定 **5 秒**（Exam06 是 6 秒，勿混）。  
2. **`LoadingRoutine` 协程**：每帧 `elapsed += Time.deltaTime`，算 `t = elapsed/loadDuration`，写 **Slider.value = t**。  
3. **TipTextTop** 显示百分比；**TipTextBottom** 按 `t` 区间换提示句（考点 4、6）。  
4. **可选 P+1**：`backgroundImage.sprite` 按 `t` 切 `Sprite[]`（考点 7）。  
5. **`t >= 1`** 后 `SceneManager.LoadScene("Exam05_GameScene")`（3.3）。

#### 场景操作

| # | 操作 |
|:-:|------|
| 1 | `Exam05/Scripts/LoadingController.cs` |
| 2 | 挂到 **`LoadingCanvas`** 或空物体 **`LoadingManager`** |
| 3 | Inspector 拖：`progressSlider` · `tipTextTop` · `tipTextBottom` · `backgroundImage`（可选 `bgSprites[]`） |

#### 带注释 P-L1 范本

```csharp
using System.Collections;
using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;

namespace Exam.Exam05
{
    /// <summary>
    /// Exam05 Loading：约 5 秒内协程刷进度条与提示，结束后进 Game 场景。
    /// </summary>
    public class LoadingController : MonoBehaviour
    {
        // --- UI：进度条 + 上下提示 + 背景（可选切图） ---
        [SerializeField] private Slider progressSlider;
        [SerializeField] private Text tipTextTop;
        [SerializeField] private Text tipTextBottom;
        [SerializeField] private Image backgroundImage;
        [SerializeField] private Sprite[] backgroundSprites;

        // --- 音频：BGM + 静音钮（3.2） ---
        [SerializeField] private Button btnMute;
        [SerializeField] private AudioSource bgmSource;

        // --- 配置：本套卷面 5 秒（Exam06 为 6） ---
        [SerializeField] private float loadDuration = 5f;
        [SerializeField] private string gameSceneName = "Exam05_GameScene";

        private void Start()
        {
            progressSlider.value = 0f;
            if (btnMute != null)
                btnMute.onClick.AddListener(ToggleMute);
            if (bgmSource != null && !bgmSource.isPlaying)
                bgmSource.Play();

            StartCoroutine(LoadingRoutine());
        }

        private IEnumerator LoadingRoutine()
        {
            float elapsed = 0f;
            while (elapsed < loadDuration)
            {
                elapsed += Time.deltaTime;
                float t = Mathf.Clamp01(elapsed / loadDuration);

                progressSlider.value = t;
                tipTextTop.text = $"加载中 {Mathf.RoundToInt(t * 100)}%";
                tipTextBottom.text = GetTipByProgress(t);
                UpdateBackground(t);

                yield return null;
            }

            progressSlider.value = 1f;
            tipTextTop.text = "加载中 100%";
            SceneManager.LoadScene(gameSceneName);
        }

        private string GetTipByProgress(float t)
        {
            if (t < 0.33f) return "正在连接服务器…";
            if (t < 0.66f) return "正在加载资源…";
            return "即将进入游戏…";
        }

        private void UpdateBackground(float t)
        {
            if (backgroundImage == null || backgroundSprites == null || backgroundSprites.Length == 0)
                return;
            int index = Mathf.Min((int)(t * backgroundSprites.Length), backgroundSprites.Length - 1);
            backgroundImage.sprite = backgroundSprites[index];
        }

        private void ToggleMute()
        {
            if (bgmSource != null)
                bgmSource.mute = !bgmSource.mute;
        }

        private void OnDestroy()
        {
            if (btnMute != null)
                btnMute.onClick.RemoveListener(ToggleMute);
        }
    }
}
```

```csharp
// P-L1+：5s 下限 + Async 预加载 · 进度取 **Min** = 两路都到位才涨条（lixiaotong 自写）
IEnumerator LoadingRoutineHybrid()
{
    var op = SceneManager.LoadSceneAsync(gameSceneName);
    op.allowSceneActivation = false;
    float elapsed = 0f;
    while (elapsed < loadDuration)
    {
        elapsed += Time.deltaTime;
        float fakeT = elapsed / loadDuration;
        float realT = op.progress / 0.9f;
        // Min：Async 快也不能提前满条 · 时间到也不能掩盖加载未完成
        progressSlider.value = Mathf.Clamp01(Mathf.Min(fakeT, realT));
        yield return null;
    }
    progressSlider.value = 1f;
    op.allowSceneActivation = true;
}
```

```csharp
using UnityEngine;
using UnityEngine.UI;

namespace Exam.Exam05
{
    /// <summary>
    /// Exam05 战斗：四攻击钮扣血 20/30/40/50，hp≤0 显示 GameOverPanel 并 Log。
    /// </summary>
    public class BattleController : MonoBehaviour
    {
        // --- UI · 场景拖引用 ---
        [SerializeField] private Slider hpSlider;
        [SerializeField] private Text hpLabel;
        [SerializeField] private GameObject gameOverPanel;

        // --- 攻击钮 · 命名方法绑定（OnDestroy 可 RemoveListener）---
        [SerializeField] private Button btnAttack20;
        [SerializeField] private Button btnAttack30;
        [SerializeField] private Button btnAttack40;
        [SerializeField] private Button btnAttack50;

        // --- 战斗参数 ---
        [SerializeField] private int maxHp = 100;

        private int currentHp;
        /// <summary>游戏结束后忽略后续扣血输入。</summary>
        private bool isGameOver;

        /// <summary>
        /// 思路：初始化 HP 与 Slider → 隐藏结束 Panel → 绑定四钮命名回调。
        /// </summary>
        private void Start()
        {
            currentHp = maxHp;
            hpSlider.maxValue = maxHp;
            hpSlider.value = currentHp;
            gameOverPanel.SetActive(false); // 卷面：Panel 默认不参与显示
            RefreshHpText();

            btnAttack20.onClick.AddListener(OnAttack20);
            btnAttack30.onClick.AddListener(OnAttack30);
            btnAttack40.onClick.AddListener(OnAttack40);
            btnAttack50.onClick.AddListener(OnAttack50);
        }

        /// <summary>思路：20 伤入口 → 转 TakeDamage。</summary>
        private void OnAttack20() => TakeDamage(20);

        /// <summary>思路：30 伤入口 → 转 TakeDamage。</summary>
        private void OnAttack30() => TakeDamage(30);

        /// <summary>思路：40 伤入口 → 转 TakeDamage。</summary>
        private void OnAttack40() => TakeDamage(40);

        /// <summary>思路：50 伤入口 → 转 TakeDamage。</summary>
        private void OnAttack50() => TakeDamage(50);

        /// <summary>
        /// 思路：门闩判断 → Clamp 扣血 → 刷新 UI → hp≤0 则结束态。
        /// </summary>
        private void TakeDamage(int damage)
        {
            if (isGameOver) return; // 已结束不再响应扣血
            currentHp = Mathf.Clamp(currentHp - damage, 0, maxHp); // P+1：钳在 [0, maxHp]
            hpSlider.value = currentHp;
            RefreshHpText();
            if (currentHp <= 0)
            {
                isGameOver = true;
                gameOverPanel.SetActive(true);
                Debug.Log("游戏结束");
            }
        }

        /// <summary>思路：同步 Slider 旁文字（hpLabel 可选）。</summary>
        private void RefreshHpText()
        {
            if (hpLabel != null)
                hpLabel.text = "生命值：" + currentHp;
        }

        /// <summary>
        /// 思路：销毁时 RemoveListener，避免重复进场景委托残留。
        /// </summary>
        private void OnDestroy()
        {
            if (btnAttack20 != null) btnAttack20.onClick.RemoveListener(OnAttack20);
            if (btnAttack30 != null) btnAttack30.onClick.RemoveListener(OnAttack30);
            if (btnAttack40 != null) btnAttack40.onClick.RemoveListener(OnAttack40);
            if (btnAttack50 != null) btnAttack50.onClick.RemoveListener(OnAttack50);
        }
    }
}
```
